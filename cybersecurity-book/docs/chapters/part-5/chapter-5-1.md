```python
        """–ü—Ä–æ–≤–µ—Ä–∏—Ç—å TOTP —Ç–æ–∫–µ–Ω"""
```

```python
def encrypt_data(plaintext: bytes, key: bytes = None) -> tuple[bytes, bytes, bytes]:
    """
    –®–∏—Ñ—Ä–æ–≤–∞–Ω–∏–µ —Å –∞—É—Ç–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ü–∏–µ–π (AES-256-GCM)

    """

        key = AESGCM.generate_key(bit_length=256)
    
    aesgcm = AESGCM(key)
    nonce = os.urandom(12)  # 96 –±–∏—Ç –¥–ª—è GCM
    ciphertext = aesgcm.encrypt(nonce, plaintext, associated_data=None)
    
    return key, nonce, ciphertext

def decrypt_data(key: bytes, nonce: bytes, ciphertext: bytes) -> bytes:
    """–î–µ—à–∏—Ñ—Ä–æ–≤–∞–Ω–∏–µ –∏ –≤–µ—Ä–∏—Ñ–∏–∫–∞—Ü–∏—è —Ü–µ–ª–æ—Å—Ç–Ω–æ—Å—Ç–∏"""
    aesgcm = AESGCM(key)
    # –ï—Å–ª–∏ –¥–∞–Ω–Ω—ã–µ –±—ã–ª–∏ –∏–∑–º–µ–Ω–µ–Ω—ã ‚Äî –≤—ã–±—Ä–æ—Å–∏—Ç InvalidTag –∏—Å–∫–ª—é—á–µ–Ω–∏–µ
    return aesgcm.decrypt(nonce, ciphertext, associated_data=None)

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ

decrypted = decrypt_data(key, nonce, ciphertext)
print(f"Decrypted: {decrypted}")
```
